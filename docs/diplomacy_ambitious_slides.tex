%% ============================================================
%% SLIDE: Key Findings
%% ============================================================
\begin{frame}{Key Findings}

\begin{enumerate}
    \item \textcolor{neutral}{\textbf{$\rho$ dominates}}: Preference correlation is the primary driver of outcomes (3$\times$ SW range). $\theta$ has negligible effect.

    \vspace{0.3cm}
    \item \textcolor{good}{\textbf{No exploitation gap}}: Unlike co-funding, stronger models do NOT extract more value. Utility gaps are $<$0.05 across all conditions.

    \vspace{0.3cm}
    \item \textcolor{highlight}{\textbf{Capability floor matters}}: GPT-3.5 (Elo 1105) breaks negotiation for \textbf{both} parties. Below Elo $\sim$1200, models can't follow the protocol.

    \vspace{0.3cm}
    \item \textbf{No order effects}: Speaking first provides no advantage. Strong and weak agents get symmetric outcomes.

    \vspace{0.3cm}
    \item \textbf{High consensus rate}: 89\% overall, rising to 99\% at $\rho{=}1$. Diplomacy's propose-and-vote protocol is robust.
\end{enumerate}
\end{frame}

%% ============================================================
%% SLIDE: Comparison with Co-Funding
%% ============================================================
\begin{frame}{Diplomacy vs Co-Funding: Contrasting Exploitation Patterns}
\begin{center}
\begin{tabular}{lcc}
\toprule
Property & Diplomacy & Co-Funding \\
\midrule
Protocol & Propose-and-vote & Talk-pledge-revise \\
Exploitation gap & \textcolor{good}{$<$ 0.05 (none)} & \textcolor{highlight}{Significant (free-riding)} \\
Dominant parameter & $\rho$ (correlation) & $\alpha$ (alignment) \\
Secondary parameter & $\theta$ (no effect) & $\sigma$ (scarcity) \\
Consensus rate & 89\% & N/A (no voting) \\
Coordination failure & Low & 77.5\% \\
Speaking order effect & None & N/A \\
\bottomrule
\end{tabular}
\end{center}

\vspace{0.3cm}
\textbf{Hypothesis}: The propose-and-vote protocol \textbf{protects} the weak agent ---
both parties must agree, so exploitation is structurally prevented.

In co-funding, each agent independently pledges funds, creating an opening for free-riding that propose-and-vote eliminates.
\end{frame}
